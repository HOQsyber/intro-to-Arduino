02المكوّنات

تعرّف على القطع

هذي القطع اللي راح نستخدمها اليوم. اضغط على أي قطعة لتعرف عنها أكثر.

لوحة الأردوينو UNO

عقل المشروع. لوحة إلكترونية فيها معالج صغير، تستقبل الأوامر اللي نكتبها بالكود وتنفذها.

  • منافذ رقمية (Digital) من 0 إلى 13
  • منافذ تماثلية (Analog) من A0 إلى A5
  • توصل بالكمبيوتر عبر كابل USB
البريدبورد

لوحة ذات منافذ، نوصل فيها القطع بدون لحام.

  • المنافذ في الصفوف موصولة ببعض داخلياً
  • تقدر تشوف مخطط توضيحي هنا الرابط
الأنوار (LED)

نور صغير، يشتغل لما يمر فيه تيار كهربائي.

  • الطرف الطويل (+) ← يوصل بالمنفذ الرقمي
  • دائما الطرف الطويل (+) في أغلب القطع
  • الطرف القصير (−) ← يوصل بـ GND
  • ⚠️ لو عكست التوصيل، الـ LED ما راح يضيء
الجرس (Buzzer)

مكوّن يصدر صوت تنبيه. وهو نوعين:

  • Active ← يصدر صوت بوجود طاقة
  • Passive ← يحتاج إشارة تردد حتى يصدر صوت
حساس المسافة HC-SR04

حساس يقيس المسافة باستخدام الموجات فوق الصوتية (مثل الخفافيش 🦇).

  • VCC ← الكهرباء (5V)
  • GND ← الأرضي
  • Trig ← يرسل الموجة
  • Echo ← يستقبل الموجة الراجعة
المقاومات (Resistors)

قطعة تتحكم في كمية التيار اللي يمر في الدائرة.

  • تحمي القطع من التيار العالي
  • نستخدم 220Ω أو 330Ω مع الأنوار
أسلاك التوصيل

أسلاك ملونة للتوصيل بين القطع. وأنواعهم:

  • Male to Male
  • Male to Female
  • Female to Female

› شرح الأوامر

الأمرماذا يفعل
pinMode(ledPin, OUTPUT)نحدد للأردوينو إن المنفذ هذا للإخراج
digitalWrite(ledPin, HIGH)نطلع 5 فولت ← الـ LED يضيء
digitalWrite(ledPin, LOW)نطلع 0 فولت ← الـ LED ينطفي
delay(1000)انتظر 1000 ميلي ثانية = ثانية واحدة
نقطة فحص
الـ LED لازم يضيء وينطفي كل ثانية بشكل متكرر. لو ما اشتغل، شوف قسم حل المشاكل.
جرّب بنفسك
  • غيّروا delay(1000) إلى delay(200) — شوفوا الفرق
  • خلوا التشغيل 100ms والإطفاء 1000ms
  • جربوا تضيفون LED ثاني على منفذ مختلف
  • جربوا تغيرون المقاومة وتشوفون تأثيرها على شدة الإضاءة

++C
#define trigPin 9
#define echoPin 10

long duration;
int  distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");
  delay(200);
}
كيف نشوف القيم؟

افتحوا Tools → Serial Monitor (Ctrl+Shift+M) وتأكدوا إن السرعة 9600 baud.

نقطة فحص
الأرقام تتغير لما تقربون أو تبعدون شي عن الحساس.
جرّب بنفسك
  • قيسوا المسافة بين الحساس وكتاب، ثم تأكدوا بالمسطرة
  • غيّروا delay(200) إلى delay(50) ← القراءات أسرع
  • النطاق الي يقراه المستشعر يختلف من نوع إلى آخر حسب جودة ومواد التصنيع

Echo (الحساس)10

LED أحمر (+)5 مع مقاومة
Buzzer (+)7
كل الـ (−)GND
VCC الحساس5V

› الكود الكامل

++C
#define trigPin  9
#define echoPin  10
#define greenLED 6
#define redLED   5
#define buzzer   7

long duration;
int  distance;

void setup() {
  pinMode(trigPin,  OUTPUT);
  pinMode(echoPin,  INPUT);
  pinMode(greenLED, OUTPUT);
  pinMode(redLED,   OUTPUT);
  pinMode(buzzer,   OUTPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  if (distance <= 15) {
    digitalWrite(redLED,   HIGH);
    digitalWrite(greenLED, LOW);
    digitalWrite(buzzer,   HIGH);
  } else {
    digitalWrite(redLED,   LOW);
    digitalWrite(greenLED, HIGH);
    digitalWrite(buzzer,   LOW);
  }
  delay(200);
}
نقطة فحص
  • يد بعيدة ← النور الأخضر مضيء، الجرس صامت
  • يد قريبة (أقل من 15 سم) ← النور الأحمر يضيء، الجرس يصوت
مبروك!

هذا أول مشروع ذكي تسوونه! اللي بنيتوه هو نفس مبدأ أجهزة الإنذار الحقيقية.

جرّب بنفسك
  • غيّروا المسافة من 15 إلى 30 سم
  • خلوا الجرس يشتغل بشكل متقطع بدل ما يكون مستمر
  • ضيفوا نور ثالث (أصفر) للمسافة المتوسطة (بين 15 و30 سم)

Reference Table

06الدعم الفني

حل المشاكل الشائعة

الكود ما يرفع للأردوينو
  • تأكدوا من اختيار اللوحة: Tools → Board → Arduino UNO
  • تأكدوا من اختيار المنفذ: Tools → Port
  • افصلوا كابل USB وأعيدوا توصيله
  • جربوا منفذ USB آخر
  • أعيدوا تشغيل البرنامج
الـ LED ما يضيء
  • تأكدوا من اتجاه الـ LED: الطرف الطويل (+) للمنفذ الرقمي
  • تأكدوا من وجود مقاومة في الدائرة
  • تأكدوا إن رقم المنفذ في الكود مطابق للتوصيل الفعلي
  • جربوا LED آخر، يمكن محترق
الـ Buzzer ما يصدر صوت
  • تأكدوا من اتجاه التوصيل: (+) للمنفذ الرقمي، (−) للـ GND
  • تأكدوا إنه Active Buzzer
  • تأكدوا من تعريف المنفذ في setup()
القيم ما تطلع في الـ Serial Monitor
  • تأكدوا من السرعة: 9600 baud
  • تأكدوا إن Serial.begin(9600); موجودة في setup()
  • أغلقوا وافتحوا الـ Serial Monitor مجدداً
قيم الحساس غير منطقية
  • تأكدوا من توصيل VCC بـ 5V وليس 3.3V
  • تأكدوا من توصيل Trig و Echo على المنافذ الصحيحة
  • بعدوا الحساس عن الأسطح الناعمة (الأقمشة تمتص الموجات)
خطأ "expected ';' before ..."

نسيتوا فاصلة منقوطة ; في آخر أحد الأسطر. ارجعوا للسطر اللي قبل اللي طلع فيه الخطأ.
